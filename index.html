Сложение и Вычитание в C++

#include <iostream>
using namespace std;
					
int main() {
int number1;
int number2;
int summa, raznost;
												
cout << "Введите значение чисел для вычисления суммы" << endl;
cin >> number1 >> number2;
												
summa = number1 + number2;
												
cout << "Результат сложения числе равен:  " << summa << endl;
												
cout << "Введите значение чисел для вычисления разности чисел" << endl;
cin >> number1 >> number2;
												
raznost = number1 - number2;
												
cout << "Результат вычитания числе равен:  " << raznost << endl;

return 0;
}
